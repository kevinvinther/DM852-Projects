133 Node *next() {
134 if (right != nullptr) {
135 Node *next = right;
136 // In a binary search tree the right-side is larger, and we want to
137 // find the smallest number that is still larger. Therefore we first
138 // go to the right, and keep going left until we get the smallest
139 // larger number.
140 while (next->left != nullptr) {
141 next = next->left;
142 }
143 return next;
144 } else { // The right node does not exist
145 Node *next = this;
146 while (next->parent) {
147 if (next != next->parent->right) {
148 return next->parent;
149 }

440 bool iteratorTraversal(const Tree &other) const {
441 if (node_count != other.node_count) {
442 return false;
443 }
444 for (const_iterator it = begin(), otherIt = other.begin();
445 it != end() && otherIt != other.end(); ++it, ++otherIt) {
446 if (!(!comp(it->first, otherIt->first) &&
447 !comp(otherIt->first, it->first)) ||
448 it->second != otherIt->second) {
449 return false;
450 }
451 }
452 return true;
